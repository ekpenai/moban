三、解析器（核心代码）
function parseFields(template) {
  const regex = /【(.*?)】/g
  const result = {}

  let match

  while ((match = regex.exec(template))) {
    const raw = match[1] // groom:text|李明轩

    const [left, value] = raw.split('|')
    const [key, type] = left.split(':')

    result[key] = {
      key,
      type: type || 'text',
      value: value || '',
      label: key
    }
  }

  return result
}
四、自动表单生成（核心）
function generateForm(fields) {
  return Object.values(fields).map(field => {
    return {
      key: field.key,
      label: field.label,
      type: field.type,
      value: field.value
    }
  })
}

五、小程序表单渲染（关键）
WXML 示例：
<view wx:for="{{form}}" wx:key="key">

  <view class="label">{{item.label}}</view>

  <input
    value="{{item.value}}"

</view>
JS 更新值：
Page({
  data: {
    form: []
  },

  onInput(e) {
    const key = e.currentTarget.dataset.key
    const value = e.detail.value

    const form = this.data.form.map(item => {
      if (item.key === key) {
        item.value = value
      }
      return item
    })

    this.setData({ form })
  }
})

七、渲染到海报（核心逻辑）
替换字段：
function renderTemplate(template, fields) {
  return template.replace(/【(.*?)】/g, (_, raw) => {
    const [left] = raw.split('|')
    const [key] = left.split(':')

    return fields[key]?.value || ''
  })
}
